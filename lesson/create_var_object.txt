// Declare an object variable using object literal notation
var person = {
  name: "John",
  age: 30,
  city: "New York"
};

// Access properties of the object
console.log(person.name); // Output: John
console.log(person.age); // Output: 30
console.log(person.city); // Output: New York

// Modify properties of the object
person.age = 35;
person.city = "San Francisco";

// Add new properties to the object
person.country = "USA";

// Access modified properties and new properties
console.log(person.age); // Output: 35
console.log(person.city); // Output: San Francisco
console.log(person.country); // Output: USA
