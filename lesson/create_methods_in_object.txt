var person = {
  name: "John",
  age: 30,
  city: "New York",
  sayHello: function() {
    console.log("Hello, my name is " + this.name);
  },
  getFullName: function() {
    return this.name + " Doe";
  }
};

person.sayHello(); // Output: Hello, my name is John

var fullName = person.getFullName();
console.log(fullName); // Output: John Doe
